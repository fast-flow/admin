<script type="text/javascript">
function copy2filter ($clone) {
    var length = $('.js-copy2').length
    var copy2filterCount = length + 1
    if (length > 4) {
        alert('最多不能超过5步骤')
        return false
    }
    $clone.find('.js-copy2-number').html(copy2filterCount)
}
</script>
