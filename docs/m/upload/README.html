<script>
function filtermax6 ($clone) {
    var length = $('.js-max6-clone').length
    // 设置编号
    $clone.attr('data-upload-title', length + 1)
    if (length > 5) {
        alert('最多上传6张图')
        return false
    }
}
</script>
<hr>
<button type="submit" class="mo-btn mo-btn--info">提交</button>
</form>
